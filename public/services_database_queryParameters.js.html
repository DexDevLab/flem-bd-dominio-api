        WHEN r.demissao IS NULL
            THEN 'N' 
        ELSE 'S' 
    END desligado

/**
 * Parâmetro SQL que retorna o estado civil do
 * funcionário baseado no código do estado civil.
 */
export const estadoCivil = `
    CASE estado_civil
        WHEN 'S' 
            THEN 'Solteiro'
        WHEN 'C' 
            THEN 'Casado'
        WHEN 'V' 
            THEN 'Viúvo'
        WHEN 'D' 
            THEN 'Divorciado'
        WHEN 'O' 
            THEN 'Concubinato'
        WHEN 'J' 
            THEN 'Separado judicialmente'
        WHEN 'U' 
            THEN 'União estável' 
        ELSE 'Não-declarado' 
    END AS estadoCivil
`;
